public class MappedRequestInfo {

    private String name;
    private String url;

    private String description;

    public MappedRequestInfo(String name, String url) {
        this.name = name;
        this.url = url;
    }

    public MappedRequestInfo(String name, String url, String description) {
        this(name, url);
        this.description = description;
    }

    public String getDescription() {
        return description;
    }

    public void setDescription(String description) {
        this.description = description;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public String getUrl() {
        return url;
    }

    public void setUrl(String url) {
        this.url = url;
    }

    @Override
    public int hashCode() {
        return url.hashCode();
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;

        MappedRequestInfo that = (MappedRequestInfo) o;

        return url.equals(that.url);

    }

    @Override
    public String toString() {
        return "MappedRequestInfo{" +
                "name='" + name + '\'' +
                ", url='" + url + '\'' +
                ", description='" + description + '\'' +
                '}';
    }
}
