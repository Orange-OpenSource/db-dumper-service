public class URICheck {

    public static boolean isUri(String possibleUri) {
        try {
            URI uri = new URI(possibleUri);
            if (uri.getScheme() == null || uri.getScheme().isEmpty()) {
                return false;
            }
        } catch (URISyntaxException e) {
            return false;
        }
        return true;
    }
}
